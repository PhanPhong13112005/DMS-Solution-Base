workspace "Dormitory Management System (DMS)" "Kiến trúc Microservices cho hệ thống quản lý Ký túc xá - Đồ án cuối kỳ" {

    model {
        // 1. Định nghĩa người dùng (Actors)
        admin = person "Ban quản lý KTX" "Người quản lý các hoạt động, cơ sở vật chất và sinh viên."
        student = person "Sinh viên" "Người thuê phòng và sử dụng dịch vụ tại KTX."

        // 2. Định nghĩa Hệ thống chính
        dms = softwareSystem "Dormitory Management System" "Hệ thống quản lý tổng thể Ký túc xá." {
            
            // Frontend
            webApp = container "Web / Mobile App" "Cung cấp giao diện trực quan cho Ban quản lý và Sinh viên thao tác." "React/Vue/Flutter" "Web Browser"
            
            // Ocelot API Gateway (BỔ SUNG MỚI - CHUẨN MICROSERVICES)
            apiGateway = container "Ocelot API Gateway" "Cửa ngõ duy nhất, định tuyến request và xác thực JWT token trước khi cho phép vào hệ thống." "ASP.NET Core"

            // Backend Services (3 Nhóm)
            roomService = container "Room & Building Service" "Quản lý tòa nhà, phòng, giường và tiện ích (Nhóm 1)." "ASP.NET Core Web API"
            studentService = container "Student & Contract Service" "Quản lý sinh viên và hợp đồng lưu trú (Nhóm 2)." "ASP.NET Core Web API"
            billingService = container "Billing & Maintenance Service" "Quản lý hóa đơn điện/nước và bảo trì (Nhóm 3)." "ASP.NET Core Web API"
            
            // Message Broker (BỔ SUNG MỚI - CHUẨN EVENT-DRIVEN)
            messageBroker = container "Message Broker / Event Bus" "Điều phối các luồng sự kiện bất đồng bộ giữa các services (tránh dính chặt code)." "RabbitMQ / Kafka" "MessageBroker"

            // Databases (Cơ sở dữ liệu độc lập)
            roomDb = container "Room DB" "Lưu trữ dữ liệu tòa nhà, sơ đồ phòng." "SQL Server" "Database"
            studentDb = container "Student DB" "Lưu trữ dữ liệu sinh viên, thông tin hợp đồng." "SQL Server" "Database"
            billingDb = container "Billing DB" "Lưu trữ dữ liệu hóa đơn, lịch sử thanh toán." "SQL Server" "Database"
        }

        // 3. Định nghĩa các mối quan hệ (Relationships)
        // User -> Frontend
        admin -> webApp "Sử dụng để quản lý vận hành"
        student -> webApp "Sử dụng để xem thông tin, thanh toán"
        
        // Frontend -> API Gateway (MỌI REQUEST PHẢI ĐI QUA ĐÂY)
        webApp -> apiGateway "Gửi tất cả request HTTP" "JSON/HTTPS"
        
        // API Gateway -> Backend Services (Định tuyến)
        apiGateway -> roomService "Định tuyến API quản lý phòng" "REST"
        apiGateway -> studentService "Định tuyến API quản lý hợp đồng" "REST"
        apiGateway -> billingService "Định tuyến API thanh toán & bảo trì" "REST"
        
        // Backend -> Database (Database Per Service)
        roomService -> roomDb "Đọc và ghi dữ liệu" "Entity Framework Core"
        studentService -> studentDb "Đọc và ghi dữ liệu" "Entity Framework Core"
        billingService -> billingDb "Đọc và ghi dữ liệu" "Entity Framework Core"
        
        // Giao tiếp Service -> Service (Đồng bộ - REST API)
        studentService -> roomService "Lấy ID phòng trống để xếp chỗ" "REST API"
        billingService -> roomService "Kiểm tra ID phòng để điều phối bảo trì" "REST API"
        
        // Giao tiếp Service -> Service (Bất đồng bộ - Event Driven)
        studentService -> messageBroker "Phát sự kiện (Publish) contract.created khi duyệt xong hợp đồng" "AMQP"
        billingService -> messageBroker "Lắng nghe (Subscribe) contract.created để tự động tạo hóa đơn thu tiền" "AMQP"
    }

    views {
        // View Level 1: System Context
        systemContext dms "SystemContext" "Sơ đồ ngữ cảnh hệ thống (Level 1)" {
            include *
            autoLayout
        }

        // View Level 2: Containers
        container dms "Containers" "Sơ đồ kiến trúc Container - Microservices (Level 2)" {
            include *
            autoLayout
        }

        // 4. Tùy chỉnh màu sắc, hình dáng (Styles)
        styles {

                background #08427b

                shape Person

                shape RoundedBox
            }
            element "Database" {
                shape Cylinder

                color #ffffff
            }
            element "Web Browser" {
                shape WebBrowser
            }
            element "MessageBroker" {
                shape Pipe
                background #f29c1f
                color #ffffff
            }
        }
    }
}
